### The Gross Margin Model

Freight brokers make money on the spread between what shippers pay and what carriers receive.

- Fixed or index-based pricing

    const flushTable = () => {
      if (tableBuffer.length < 2) { tableBuffer = []; return; }
      const headers = tableBuffer[0].split('|').map(h => h.trim()).filter(Boolean);
      const rows = tableBuffer.slice(2).map(r => r.split('|').map(c => c.trim()).filter(Boolean));
      elements.push(
        <div key={`table-${i}`} style={{ overflowX: 'auto', margin: '16px 0' }}>
          <table style={{ borderCollapse: 'collapse', width: '100%', fontSize: '14px' }}>
            <thead>
              <tr>
                {headers.map((h, hi) => (
                  <th key={hi} style={{ background: '#f1f5f9', padding: '8px 12px', textAlign: 'left', borderBottom: '2px solid #cbd5e1', fontWeight: '600', whiteSpace: 'nowrap' }}>{h}</th>
                ))}
              </tr>
            </thead>
            <tbody>
              {rows.map((row, ri) => (
                <tr key={ri} style={{ borderBottom: '1px solid #e2e8f0', background: ri % 2 === 0 ? '#fff' : '#f8fafc' }}>
                  {row.map((cell, ci) => (
                    <td key={ci} style={{ padding: '8px 12px' }}>{cell}</td>
                  ))}
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      );
      tableBuffer = [];
    };

    while (i < lines.length) {
      const line = lines[i];

      if (line.trim().startsWith('|')) {
        inTable = true;
        tableBuffer.push(line);
        i++;
        continue;
      } else if (inTable) {
        flushTable();
        inTable = false;
      }

      if (line.startsWith('#### ')) {
        elements.push(<h4 key={i} style={{ fontWeight: '700', fontSize: '15px', color: '#1e293b', marginTop: '20px', marginBottom: '6px' }}>{line.slice(5)}</h4>);
      } else if (line.startsWith('### ')) {
        elements.push(<h3 key={i} style={{ fontWeight: '700', fontSize: '17px', color: '#1e293b', marginTop: '24px', marginBottom: '8px', borderBottom: '1px solid #e2e8f0', paddingBottom: '4px' }}>{line.slice(4)}</h3>);
      } else if (line.startsWith('## ')) {
        elements.push(<h2 key={i} style={{ fontWeight: '800', fontSize: '20px', color: '#0f172a', marginTop: '8px', marginBottom: '12px' }}>{line.slice(3)}</h2>);
      } else if (line.startsWith('> ')) {
        elements.push(
          <blockquote key={i} style={{ borderLeft: '4px solid #f59e0b', paddingLeft: '16px', margin: '16px 0', background: '#fffbeb', padding: '12px 16px', borderRadius: '0 8px 8px 0', fontStyle: 'italic', color: '#78350f' }}>
            {line.slice(2)}
          </blockquote>
        );
      } else if (line.match(/^\d+\. /)) {
        const listItems = [];
        while (i < lines.length && lines[i].match(/^\d+\. /)) {
          const itemText = lines[i].replace(/^\d+\. /, '');
          listItems.push(<li key={i} style={{ marginBottom: '6px', lineHeight: '1.6' }} dangerouslySetInnerHTML={{ __html: itemText.replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>') }} />);
          i++;
        }
        elements.push(<ol key={`ol-${i}`} style={{ paddingLeft: '24px', margin: '12px 0' }}>{listItems}</ol>);
        continue;
      } else if (line.startsWith('- ') || line.startsWith('* ') || line.startsWith('✅') || line.startsWith('📄') || line.startsWith('🔖') || line.startsWith('☐')) {
        const listItems = [];
        while (i < lines.length && (lines[i].startsWith('- ') || lines[i].startsWith('* ') || lines[i].startsWith('✅') || lines[i].startsWith('📄') || lines[i].startsWith('🔖') || lines[i].startsWith('☐'))) {
          const raw = lines[i].replace(/^[-*] /, '');
          listItems.push(<li key={i} style={{ marginBottom: '4px', lineHeight: '1.6' }} dangerouslySetInnerHTML={{ __html: raw.replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>') }} />);
          i++;
        }
        elements.push(<ul key={`ul-${i}`} style={{ paddingLeft: '24px', margin: '10px 0' }}>{listItems}</ul>);
        continue;
      } else if (line.startsWith('---')) {
        elements.push(<hr key={i} style={{ border: 'none', borderTop: '1px solid #e2e8f0', margin: '20px 0' }} />);
      } else if (line.trim() === '') {
        elements.push(<div key={i} style={{ height: '8px' }} />);
      } else if (line.startsWith('```')) {
        const codeLines = [];
        i++;
        while (i < lines.length && !lines[i].startsWith('```')) {
          codeLines.push(lines[i]);
          i++;
        }
        elements.push(
          <pre key={i} style={{ background: '#1e293b', color: '#e2e8f0', padding: '16px', borderRadius: '8px', overflowX: 'auto', fontSize: '13px', margin: '16px 0', lineHeight: '1.7' }}>
            {codeLines.join('\n')}
          </pre>
        );
      } else {
        const html = line
          .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
          .replace(/\*(.*?)\*/g, '<em>$1</em>');
        elements.push(<p key={i} style={{ margin: '8px 0', lineHeight: '1.75', color: '#374151' }} dangerouslySetInnerHTML={{ __html: html }} />);
      }
      i++;
    }

    if (inTable) flushTable();
    return elements;
  };

  return <div style={{ fontFamily: 'Georgia, serif' }}>{renderContent(content)}</div>;
}

                  </button>
                )}
                {currentLessonIdx < totalLessons - 1 && (
                  <button
                    onClick={() => { markComplete(); nextLesson(); }}
                    style={{ background: activeModule.color, color: '#fff', border: 'none', padding: '12px 24px', borderRadius: '10px', cursor: 'pointer', fontSize: '14px', fontWeight: '700', boxShadow: `0 4px 12px ${activeModule.color}40` }}
                  >
                    Next Lesson →
                  </button>
                )}
                {currentLessonIdx === totalLessons - 1 && (
                  <button
                    onClick={() => { markComplete(); setActiveView('home'); }}
                    style={{ background: '#16a34a', color: '#fff', border: 'none', padding: '12px 24px', borderRadius: '10px', cursor: 'pointer', fontSize: '14px', fontWeight: '700' }}
                  >
                    🎉 Finish Course
                  </button>
                )}
              </div>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
